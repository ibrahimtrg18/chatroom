# chatroom
chatroom with React, NodeJS, MongoDB, Socket-IO

# first, you need to install all the node_modules:
cd chatroom

# for client:
cd client/

npm install

# for server:
cd server/

npm install

# after that, start the server and client.

# for client:
cd client/

npm start

# for server:
cd server/

node index.js
